# Money - Adding or removing money from the target country
game_master_events.10 = {
	type = country_event
	placement = ROOT

	title = game_master_events.10.title

	event_image = { video = "unspecific_signed_contract" }
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	duration = 1

	trigger = {}

	immediate = {
		random_country = {
			limit = { has_variable = game_master_action_target }
			save_scope_as = target_country
		}
		set_global_variable = {
			name = game_master_action_money_amount
			value = 0
		}
	}

	# Adding Money
	option = {
		name = game_master_events.10.a

		trigger_event = {
			id = game_master_events.11
			days = 0
			popup = yes
		}

		set_global_variable = { name = game_master_action_money_add }
	}

	# Removing Money
	option = {
		name = game_master_events.10.b

		trigger_event = {
			id = game_master_events.11
			days = 0
			popup = yes
		}

		set_global_variable = { name = game_master_action_money_remove }
	}

	# Do nothing
	option = {
		name = game_master_events.10.c

		default_option = yes

		scope:target = {
			remove_variable = game_master_action_target
		}

		remove_global_variable = game_master_action_money_amount
		remove_global_variable = game_master_taking_action
	}
}

	# Confirm Influence Action
	option = {
		name = game_master_events.59.a

		if = {
			limit = { has_global_variable = game_master_action_influence_add }
			if = {
				limit = { has_global_variable = game_master_action_influence_add_modifier }
				scope:target_country = {
					add_modifier = {
						name = game_master_influence_add_modifier
						multiplier = global_var:game_master_action_influence_amount
						months = game_master_modifier_influence_duration
					}
					remove_variable = game_master_action_target
				}

				remove_global_variable = game_master_action_influence_add_modifier
				remove_global_variable = game_master_action_influence_amount
				remove_global_variable = game_master_action_influence_duration
			}
			else_if = {
				limit = { has_global_variable = game_master_action_influence_multiplier_modifier }
				scope:target_country = {
					add_modifier = {
						name = game_master_influence_multiplier_modifier
						multiplier = global_var:game_master_action_influence_amount
						months = game_master_modifier_influence_duration
					}
					remove_variable = game_master_action_target
				}
				
				remove_global_variable = game_master_action_influence_multiplier_modifier
				remove_global_variable = game_master_action_influence_amount
				remove_global_variable = game_master_action_influence_duration
			}
		}
		else_if = {
			limit = { has_global_variable = game_master_action_influence_remove }
			if = {
				limit = { has_global_variable = game_master_action_influence_add_modifier }
				scope:target_country = {
					add_modifier = {
						name = game_master_influence_add_modifier
						multiplier = global_var:game_master_action_influence_amount
						months = game_master_modifier_influence_duration
					}
					remove_variable = game_master_action_target
				}

				remove_global_variable = game_master_action_influence_add_modifier
				remove_global_variable = game_master_action_influence_amount
				remove_global_variable = game_master_action_influence_duration
			}
			else_if = {
				limit = { has_global_variable = game_master_action_influence_multiplier_modifier }
				scope:target_country = {
					add_modifier = {
						name = game_master_influence_multiplier_modifier
						multiplier = global_var:game_master_action_influence_amount
						months = game_master_modifier_influence_duration
					}
					remove_variable = game_master_action_target
				}
				
				remove_global_variable = game_master_action_influence_multiplier_modifier
				remove_global_variable = game_master_action_influence_amount
				remove_global_variable = game_master_action_influence_duration
			}
		}
	}

	# Reset Influence Action
	option = {
		name = game_master_events.59.b

		remove_global_variable = game_master_action_influence_amount
		remove_global_variable = game_master_action_influence_duration

		if = {
			limit = { has_global_variable = game_master_action_influence_add }
			remove_global_variable = game_master_action_influence_add
		}
		if = {
			limit = { has_global_variable = game_master_action_influence_remove }
			remove_global_variable = game_master_action_influence_remove
		}
		if = {
			limit = { has_global_variable = game_master_action_influence_add_modifier }
			remove_global_variable = game_master_action_influence_add_modifier
		}
		if = {
			limit = { has_global_variable = game_master_action_influence_multiplier_modifier }
			remove_global_variable = game_master_action_influence_multiplier_modifier
		}

		trigger_event = {
			id = game_master_events.50
			days = 0
			popup = yes
		}
	}

	# Go back to the Game Master Main Menu
	option = {
		name = game_master_events.59.c

		default_option = yes

		remove_global_variable = game_master_action_influence_amount
		remove_global_variable = game_master_action_influence_duration

		if = {
			limit = { has_global_variable = game_master_action_influence_add }
			remove_global_variable = game_master_action_influence_add
		}
		if = {
			limit = { has_global_variable = game_master_action_influence_remove }
			remove_global_variable = game_master_action_influence_remove
		}
		if = {
			limit = { has_global_variable = game_master_action_influence_add_modifier }
			remove_global_variable = game_master_action_influence_add_modifier
		}
		if = {
			limit = { has_global_variable = game_master_action_influence_multiplier_modifier }
			remove_global_variable = game_master_action_influence_multiplier_modifier
		}

		trigger_event = {
			id = game_master_events.1
			days = 0
			popup = yes
		}
	}
}
