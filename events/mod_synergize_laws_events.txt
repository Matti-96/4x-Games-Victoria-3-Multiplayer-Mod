namespace = power_bloc_synergize_laws_events

power_bloc_synergize_laws_events.1 = {
	type = country_event

	title = power_bloc_synergize_laws_events.1.title
	desc = power_bloc_synergize_laws_events.1.desc
	flavor = power_bloc_synergize_laws_events.1.flavor

	event_image = {
		video = "gfx/event_pictures/southamerica_aristocrats.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/southamerica/aristocrats"

	icon = "gfx/interface/icons/event_icons/event_protest.dds"

	duration = 1

	trigger = {
		# Triggered by the Power Bloc Synergize Laws Journal Entry
	}

	immediate = {
		if = {
			limit = {
				is_power_bloc_leader = no
				is_in_power_bloc = yes
				power_bloc ?= { has_identity = identity:identity_ideological_union }
			}
			power_bloc.power_bloc_leader = {
				save_scope_as = power_bloc_leader_scope
			}
		}
		
		scope:power_bloc_leader_scope.active_law:lawgroup_governance_principles.type = {
			save_scope_as = power_bloc_leader_governance_principle_law
		}
	}

	option = {
		name = power_bloc_synergize_laws_events.1.a

		default_option = yes

		activate_law = scope:power_bloc_leader_governance_principle_law

		remove_modifier = power_bloc_leader_lawgroup_governance_principles_variable_12

		add_modifier = {
			name = power_bloc_leader_lawgroup_governance_principles_law_changed
			months = 120
			is_decaying = yes
		}
	}
}
