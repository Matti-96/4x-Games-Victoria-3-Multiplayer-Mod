# List of changes:
#	- Japanese Imperial Clothes Trigger:
#		-- Culture requirement changed from Yamato to Japonic

japanese_imperial_clothes_trigger = {
	exists = scope:character
	scope:character = {
		is_ruler = yes
		exists = home_country
     	home_country = {
     		has_law = law_type:law_monarchy
     	}
		OR = {
			AND = {
				culture = {
					has_discrimination_trait = japonic
				}
				OR = {
					NOT = {
						exists = c:JAP
					}
					AND = {
						exists = c:JAP
						NOT = { home_country = c:JAP }
					}
				}
			}
			AND = {
				exists = c:JAP
				home_country = {
					this = c:JAP
					NOT = { has_variable = meiji_restoration_complete }
				}
			}
		}
	}
}
