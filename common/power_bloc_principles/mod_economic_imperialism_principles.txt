# Economic Imperialism Principles
principle_economic_imperialism_1 = {
	icon = "gfx/interface/icons/principles_icons/custom_principles_placeholder.dds"
	background = "gfx/interface/icons/principles_icons/principle_tier_1.dds"

	non_leader_modifier = {
		# Nationalization of buildings for power bloc members who aren't the power bloc leader
		country_disable_nationalization_bool = yes
	}

	power_bloc_modifier = {
		# Reduces cohesion
		power_bloc_cohesion_add = -5
	}

	ai_weight = {
		value = 100

		if = {
			limit = { num_subjects > 0 }
			add = 100
		}

		if = {
			limit = { power_bloc ?= { NOT = { has_identity = identity:identity_sovereign_empire } } }
			multiply = 0.5
		}

		if = {
			limit = { power_bloc ?= { NOT = { has_principle_group = principle_group:principle_group_vassalization } } }
			multiply = 0.5
		}
	}
}
