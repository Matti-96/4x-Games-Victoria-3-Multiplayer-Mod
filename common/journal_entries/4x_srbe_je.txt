# coal # sulfur
je_zw_coal_sulfur_research = {
	# activation
	icon = "gfx/interface/icons/invention_icons/zw_tech_coal_sulfur.dds"
	group = je_group_technology
	weight = 99999
	should_be_pinned_by_default = yes
	is_shown_when_inactive = {
		country_innovation >= zw_value_innovation_easier
		NOT = { has_technology_researched = zw_tech_coal_sulfur }
	}
	possible = {
		has_technology_researched = combustion_engine
		market = {
			custom_tooltip = {
				text = zw_is_market_owner_tt
				owner = prev
			}
			mg:coal = { market_goods_sell_orders >= zw_value_sell_orders_coal_easier }
			mg:sulfur = { market_goods_sell_orders >= zw_value_sell_orders_sulfur_easier }
			OR = {
				mg:coal = { market_goods_pricier >= 0.15 }
				mg:sulfur = { market_goods_pricier >= 0.15 }
			}
		}
	}
	# progress bar
	scripted_progress_bar = zw_coal_sulfur_research_progress_bar
	# on complete
	complete = {
		custom_tooltip = {
			text = zw_research_bar_finished_tt
			scope:journal_entry = {
				"scripted_bar_progress(zw_coal_sulfur_research_progress_bar)" >= 60
			}
		}
	}
	on_complete = {
		add_technology_researched = zw_tech_coal_sulfur
		if = {
			limit = {
				NOT = { has_journal_entry = je_zw_resource_management }
			}
			add_journal_entry = { type = je_zw_resource_management }
		}
	}
}

je_zw_resource_management = {

	icon = "gfx/interface/icons/state_status_icons/resources.dds"

	group = je_group_historical_content
	weight = 99999
	should_be_pinned_by_default = yes
	is_shown_when_inactive = {
		OR = {
			has_technology_researched = zw_tech_coal_sulfur
			has_technology_researched = zw_tech_iron_lead
			has_technology_researched = zw_tech_oil
			has_technology_researched = zw_tech_fish
			has_technology_researched = zw_tech_wood_rubber
			has_technology_researched = zw_tech_arable_land
		}
	}
	possible = { always = yes }

	immediate = {
		set_variable = {
			name = zw_var_count_1_coal_sulfur
			value = 0
		}
		set_variable = {
			name = zw_var_count_1_iron_lead
			value = 0
		}
		set_variable = {
			name = zw_var_count_1_oil
			value = 0
		}
		set_variable = {
			name = zw_var_count_1_fish
			value = 0
		}
		set_variable = {
			name = zw_var_count_2_coal_sulfur
			value = 0
		}
		set_variable = {
			name = zw_var_count_2_iron_lead
			value = 0
		}
		set_variable = {
			name = zw_var_count_2_oil
			value = 0
		}
		set_variable = {
			name = zw_var_count_2_fish
			value = 0
		}
	}

	# debt control
	scripted_button = zw_add_resource_debt_control_button
	scripted_button = zw_add_resource_debt_control_undo_button

	# + zw_var_add_coal_sulfur
	scripted_button = zw_add_coal_sulfur_button
	scripted_button = zw_add_coal_sulfur_undo_button

	# + zw_var_add_iron_lead
	scripted_button = zw_add_iron_lead_button
	scripted_button = zw_add_iron_lead_undo_button

	# + zw_var_add_oil
	scripted_button = zw_add_oil_button
	scripted_button = zw_add_oil_undo_button

	# + zw_var_add_fish
	scripted_button = zw_add_fish_button
	scripted_button = zw_add_fish_undo_button

	# + zw_var_add_coal_sulfur_subjects
	scripted_button = zw_add_coal_sulfur_subjects_button
	scripted_button = zw_add_coal_sulfur_subjects_undo_button
	# + zw_var_add_iron_lead_subject
	scripted_button = zw_add_iron_lead_subjects_button
	scripted_button = zw_add_iron_lead_subjects_undo_button
	# + zw_var_add_oil_subjects
	scripted_button = zw_add_oil_subjects_button
	scripted_button = zw_add_oil_subjects_undo_button

	# + zw_var_add_fish_subjects
	scripted_button = zw_add_fish_subjects_button
	scripted_button = zw_add_fish_subjects_undo_button

	# + zw_var_add_arable_land
	scripted_button = zw_add_arable_land_button
	scripted_button = zw_add_arable_land_undo_button

	# + zw_var_add_arable_land budget
	scripted_button = zw_add_arable_land_budget_increase_button
	scripted_button = zw_add_arable_land_budget_decrease_button

	# + zw_var_add_arable_land budget subjects
	scripted_button = zw_add_arable_land_budget_increase_subjects_button
	scripted_button = zw_add_arable_land_budget_decrease_subjects_button

	on_monthly_pulse = {
		effect = {
			if = {
				limit = { has_variable = zw_var_add_resource_progress }
				remove_variable = zw_var_add_resource_progress
			}
			zw_effect_add_resource = {
				resource_1 = coal_sulfur
			}
			zw_effect_add_resource = {
				resource_1 = iron_lead
			}
			zw_effect_add_resource = {
				resource_1 = oil
			}
			zw_effect_add_resource = {
				resource_1 = fish
			}
		}
	}

	on_yearly_pulse = {
		effect = {
			zw_effect_add_arable_land = yes
			zw_effect_add_arable_land_subjects = yes
		}
	}
}
