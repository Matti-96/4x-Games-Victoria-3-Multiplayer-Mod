zw_coal_sulfur_research_progress_bar = {
	name = "zw_coal_sulfur_research_progress_bar"
	desc = "zw_coal_sulfur_research_progress_bar_desc"
	default = yes
	start_value = 0
	min_value = 0
	max_value = 60
	monthly_progress = {
		if = {
			limit = {
				market.owner = this
				market = {
					mg:coal = { market_goods_sell_orders >= zw_value_sell_orders_coal }
					mg:sulfur = { market_goods_sell_orders >= zw_value_sell_orders_sulfur }
				}
			}
			add = {
				desc = "zw_coal_sulfur_research_progress_bar_sell_orders_tt"
				value = 1
			}
		}
		if = {
			limit = { country_innovation >= zw_value_innovation }
			add = {
				desc = "zw_coal_sulfur_research_progress_bar_innovation_tt"
				value = 1
			}
		}
		if = {
			limit = {
				market = {
					OR = {
						mg:coal = { market_goods_pricier >= 0.20 }
						mg:sulfur = { market_goods_pricier >= 0.20 }
					}
				}
			}
			add = {
				desc = "zw_coal_sulfur_research_progress_bar_goods_pricier_tt"
				value = 1
			}
		}
	}
}
