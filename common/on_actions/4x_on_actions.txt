UHM_on_action = {
	effect = {
		random_political_movement = {
			limit = {
				is_secessionist = yes
				political_movement_support = 0
			}
			add_modifier = {
				name = placated_rally_ig
				months = 2
				multiplier = 10
			}
		}
	}
}

HMPS_Check_for_GM = {
	effect = {
		if = {
			limit = {
				has_global_variable = player_gm
				any_country = {
					has_modifier = game_master
					is_player = no
				}
			}
			random_country = {
				limit = {
					has_modifier = game_master
				}
				remove_modifier = game_master
				remove_global_variable = player_gm
			}
		}
	}
}

HMPS_undo_sp = {
	effect = {
		if = {
			limit = {
				has_global_variable = single_player
				has_modifier = game_master
				any_country = {
					is_player = yes
					NOT = { this = ROOT }
				}
			}
			remove_global_variable = single_player
		}
	}
}
