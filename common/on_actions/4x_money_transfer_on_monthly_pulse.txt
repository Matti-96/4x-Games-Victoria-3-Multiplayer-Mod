on_money_transfer_monthly_pulse_country = {
	effect = {
		# Add money transfer variables to countries that doesnt have them yet
		if = {
			limit = {
				NOT = {
					has_variable = country_money_transfer_amount
				}
			}
			set_variable = {
				name = country_money_transfer_amount
				value = 0
			}
		}
	}
}
