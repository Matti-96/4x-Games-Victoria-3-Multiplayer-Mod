# coal_sulfur

zw_add_coal_sulfur_button = {
	name = "zw_add_coal_sulfur_button"
	desc = "zw_add_coal_sulfur_button_desc"
	visible = { has_technology_researched = zw_tech_coal_sulfur }
	selected = { has_variable = zw_var_add_coal_sulfur }
	possible = {}
	effect = { 
		custom_tooltip = {
			text = zw_add_coal_sulfur_button_effect_tt
			set_variable = zw_var_add_coal_sulfur
			if = {
				limit = { has_modifier = zw_modifier_add_coal_sulfur_cost }
				remove_modifier = zw_modifier_add_coal_sulfur_cost
			}
			add_modifier = {
				name = zw_modifier_add_coal_sulfur_cost
				multiplier = zw_value_money_cost_for_research
			}
			add_treasury = {
				subtract = zw_value_money_cost_for_research_up_front
			}
		}
	}
	ai_chance = {
		base = 40
		modifier = {
			trigger = {
				OR = {
					# income low
					weekly_net_fixed_income < 100000
					# gold_reserves low
					scaled_gold_reserves < 0.4
					# goods_price cheap
					market = {
						AND = {
							mg:coal = { market_goods_pricier < 0.10 }
							mg:sulfur = { market_goods_pricier < 0.10 }
						}
					}
					# can_build_no, can_build_no_subjects
					NOR = {
						any_scope_building = {
							OR = {
								is_building_type = building_coal_mine_zw
								is_building_type = building_sulfur_mine_zw
							}
							level >= 35
						}
						any_subject_or_below = {
							any_scope_building = {
								OR = {
									is_building_type = building_coal_mine
									is_building_type = building_sulfur_mine
								}
								level >= 35
							}
						}
					}
				}
			}
			add = -1000
		}
		#modifier = {
		#	trigger = { has_variable = zw_var_add_coal_sulfur }
		#	add = -10
		#}
		modifier = {
			trigger = { has_variable = zw_var_add_iron_lead }
			add = -10
		}
		modifier = {
			trigger = { has_variable = zw_var_add_oil }
			add = -10
		}
		modifier = {
			trigger = { has_variable = zw_var_add_fish }
			add = -10
		}
	}
}
