# Mitsui
company_mitsui = {
	icon = "gfx/interface/icons/company_icons/historical_company_icons/japanese_mitsui.dds"
	background = "gfx/interface/icons/company_icons/company_backgrounds/comp_illu_manufacturing_light.dds"

	flavored_company = yes

	building_types = {
		building_textile_mills
		building_paper_mills
	}

	potential = {
		has_interest_marker_in_region = region_japan
	}

	attainable = {
		hidden_trigger = { # Hide attainable if it's always fulfilled by possible
			any_scope_state = {
				state_region = s:STATE_KANSAI
			}
		}
	}

	possible = { 
		country_has_primary_culture = cu:japanese
		any_scope_state = {
			state_region = s:STATE_KANSAI
			is_incorporated = yes
			OR = {
				any_scope_building = {
					is_building_type = building_textile_mills
					level >= 10
				}
				any_scope_building = {
					is_building_type = building_paper_mills
					level >= 10
				}
			}
		}
		has_technology_researched = stock_exchange
	}

	prosperity_modifier = {
		interest_group_ig_industrialists_pol_str_mult = 0.1
		interest_group_ig_industrialists_approval_add = 1
		country_tech_spread_mult = 0.1
	}

	ai_weight = {
		value = 3 # Higher base value for flavored companies
	}
}
